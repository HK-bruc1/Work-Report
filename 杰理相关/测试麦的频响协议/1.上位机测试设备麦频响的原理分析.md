### 1.1 宏配置


## 三、数据接收与分发调用链

### 3.1 从 SPP 到 CVP DUT

```
SPP 数据到达
  online_spp_recieve_cbk(priv, buf, len)              [spp_online_db.c:92]
    db_api->packet_handle(buf, len)                     [online_db_deal.c:197]
      │
      解析外层头: struct db_head_t { u8 length; u8 type; u8 seq; u8 data[]; }
      │
      ├─ type 必须在 [1, DB_PKT_TYPE_MAX) 范围内
      ├─ 查表: db_parse_data = (void *)db_cb_api_table[type]
      │   type == 0x0B → cvp_dut_app_online_parse       ★ 在 cvp_dut_init() 时写入
      │
      ├─ size -= 3  // 去掉外层头
      ├─ memcpy(tmp_buf, db_ptr->data, size)  对齐整理
      └─ db_parse_data(tmp_buf, size, ext_data=&db_ptr->type, ext_size=2)
            │  // ext_data[0]=type(=0x0B), ext_data[1]=seq → 供回调取用
            │
            ▼
    cvp_dut_app_online_parse(packet, size, ext_data, ext_size)  [audio_cvp_dut.c:283]
      cvp_dut_hdl->parse_seq = ext_data[1]             记录 seq (ext_data[1] 即外层头的 seq 字段)
      int ret = cvp_dut_spp_rx_packet(packet, size)   ① 先尝试旧格式 55BB
      if (ret)                                         ② 失败则尝试新格式 55CC
          cvp_dut_new_rx_packet(packet, size)
```

### 3.2 旧格式处理 (0x55BB)

```c
// 数据格式: MAGIC(2B=0x55BB) + CRC16(2B) + dat[4B]，固定 8 字节
// audio_cvp_dut.c:296

cvp_dut_spp_rx_packet(dat, len)
  ├─ len > 8 → return 1 (fallback 到新格式)
  ├─ CRC16 = CRC16(dat+4, len-4)                      校验 dat[4]~dat[7]
  ├─ 大端格式: 先做字节序转换 (每 2B swap)
  ├─ memcpy → cvp_dut_hdl->rx_buf
  ├─ magic == 0x55BB ?
  └─ crc 匹配 (或 crc==0x01) ?
       memcpy 全部字段
       ret = cvp_dut_event_deal(rx_buf.dat)            ★ 命令执行
       cvp_dut_spp_tx_packet(ret)                      回复 (同样 8 字节 55BB 格式)
       return 0
```

**旧格式仅支持 dat[3] 单字节命令（0x01~0x0D），无变长参数。**

### 3.3 新格式处理 (0x55CC)

```c
// 数据格式: MAGIC(2B=0x55CC) + CRC16(2B) + CMD(1B) + LEN(2B) + DATA[]
// struct cvp_dut_head_t { u16 magic; u16 crc; struct { u8 cmd; u16 len; u8 dat[0]; } packet; }
// audio_cvp_dut.c:614

cvp_dut_new_rx_packet(dat, len)
  struct cvp_dut_head_t *head = (void *)dat
  crc = CRC16(&head->packet, len - 4)                  校验 CMD+LEN+DATA
  ├─ magic != 0x55CC → return 1
  ├─ crc 不匹配 (且 crc != 0x01) → cvp_dut_new_tx_packet(0xA2)  回复 CRC 错误
  └─ crc 匹配 →
       ret = cvp_dut_new_event_deal(&head->packet)     ★ 命令执行
       cvp_dut_new_tx_packet(ret, NULL, 0)             回复
```

## 四、命令执行调用链

### 4.1 旧格式命令分发

```c
// audio_cvp_dut.c:97
// 仅处理 dat[3] 这一个字节作为命令号

cvp_dut_event_deal(dat)
  switch (dat[3]):

    // ── 双麦 DMS ── (TCFG_AUDIO_DUAL_MIC_ENABLE)
    case CVP_DUT_DMS_MASTER_MIC (0x01):    // 主 MIC
        mode = ALGORITHM
        audio_aec_output_sel(DMS_OUTPUT_SEL_MASTER, 0)  → [audio_aec_dms.c:1135]
    case CVP_DUT_DMS_SLAVE_MIC  (0x02):    // 副 MIC(FF)
        mode = ALGORITHM
        audio_aec_output_sel(DMS_OUTPUT_SEL_SLAVE, 0)
    case CVP_DUT_DMS_OPEN_ALGORITHM (0x03): // 开启算法
        mode = ALGORITHM
        audio_aec_output_sel(DMS_OUTPUT_SEL_DEFAULT, 1)

    // ── 三麦 ── (TCFG_AUDIO_TRIPLE_MIC_ENABLE)
    case CVP_DUT_3MIC_MASTER_MIC (0x01):     audio_aec_output_sel(1, 0)
    case CVP_DUT_3MIC_SLAVE_MIC  (0x02):     audio_aec_output_sel(2, 0)
    case CVP_DUT_3MIC_FB_MIC     (0x0E):     audio_aec_output_sel(3, 0)
    case CVP_DUT_3MIC_OPEN_ALGORITHM (0x03): audio_aec_output_sel(0, 0)

    // ── 单麦 SMS ── (else)
    // 注意: 以下 case 值(0x01/0x03)与双麦/三麦相同，但位于 #else 分支，互斥编译
    case CVP_DUT_DMS_MASTER_MIC (0x01):     audio_aec_toggle_set(0)  → [audio_aec.c:704]
    case CVP_DUT_DMS_OPEN_ALGORITHM (0x03): audio_aec_toggle_set(1)

    // ── ANC MIC BYPASS ── (TCFG_AUDIO_ANC_ENABLE)
    case CVP_DUT_FF_MIC (0x06):               mode = BYPASS; cvp_dut_bypass_mic_ch_sel(ANCL_FF_MIC)
    case CVP_DUT_FB_MIC (0x07):               mode = BYPASS; cvp_dut_bypass_mic_ch_sel(ANCL_FB_MIC)
    case CVP_DUT_HEAD_PHONE_R_FF_MIC (0x08):  mode = BYPASS; cvp_dut_bypass_mic_ch_sel(ANCR_FF_MIC)
    case CVP_DUT_HEAD_PHONE_R_FB_MIC (0x09):  mode = BYPASS; cvp_dut_bypass_mic_ch_sel(ANCR_FB_MIC)

    // ── ANS/DNS 开关 ── (非 TCFG_CVP_DEVELOP_ENABLE 时编译)
    case CVP_DUT_NS_OPEN (0x04):   audio_cvp_ioctl(CVP_NS_SWITCH, 1, NULL)
    case CVP_DUT_NS_CLOSE (0x05):  audio_cvp_ioctl(CVP_NS_SWITCH, 0, NULL)

    // ── CVP 算法开关 ──
    case CVP_DUT_OPEN_ALGORITHM (0x0A):  audio_aec_toggle_set(1)
    case CVP_DUT_CLOSE_ALGORITHM (0x0B): audio_aec_toggle_set(0)

    // ── 关机 ──
    case CVP_DUT_POWEROFF (0x0D):  sys_enter_soft_poweroff(NULL)

    default: return CVP_DUT_ACK_ERR_UNKNOW (0xA1)
```

### 4.2 新格式命令分发

```c
// audio_cvp_dut.c:353
// packet 为 struct cvp_dut_packet_t { u8 cmd; u16 len; u8 dat[]; }

cvp_dut_new_event_deal(packet)
  switch (hdl->cmd):

    // 0x01~0x0D 的 MIC/算法控制 → 与旧格式相同（重写了 switch-case）
    // 此外新增以下变长命令:

    case CVP_ENC_MIC_DIFF_CMP_SET (0x0F):     // TCFG_AUDIO_MIC_ARRAY_TRIM_ENABLE
        memcpy(&diff, hdl->dat, 4)              float → diff
        dms_mic_diff = log10(app_var.enc_degradation) * 20
        app_var.audio_mic_array_diff_cmp = pow(10, (diff+dms_mic_diff)/20)
        syscfg_write(CFG_MIC_ARRAY_DIFF_CMP_VALUE, ...)
        → VM 写入失败 → return CVP_DUT_ACK_VM_WRITE_ERR (0xA3)

    case CVP_ENC_MIC_DIFF_CMP_EN (0x10):      // 补偿使能/关闭
        memcpy(&en, hdl->dat, 1)                u8: 0=关 1=开
        app_var.audio_mic_array_trim_en = en
        syscfg_write(CFG_MIC_ARRAY_TRIM_EN, ...)

    case CVP_ENC_MIC_DIFF_CMP_CLEAN (0x11):   // 清除所有补偿
        app_var.audio_mic_array_diff_cmp = 1.0f
        app_var.audio_mic_cmp.talk = ff = fb = 1.0f
        syscfg_write(CFG_MIC_ARRAY_DIFF_CMP_VALUE, ...)
        syscfg_write(CFG_MIC_TARGET_DIFF_CMP, ...)    清除金机差值也一并清零

    case CVP_ENC_TARGET_MIC_DIFF_SET (0x14):  // 写入金机 MIC 差值 (3×float)
        memcpy(&talk_diff, hdl->dat, 4)
        memcpy(&ff_diff, hdl->dat+4, 4)
        memcpy(&fb_diff, hdl->dat+8, 4)
        app_var.audio_mic_cmp.talk = pow(10, talk_diff/20)
        app_var.audio_mic_cmp.ff   = pow(10, ff_diff/20)
        app_var.audio_mic_cmp.fb   = pow(10, fb_diff/20)
        syscfg_write(CFG_MIC_TARGET_DIFF_CMP, ...)

    case CVP_DUT_CVP_IOCTL (0x12):            // 算法单元独立控制
        u8 ioctl_cmd = hdl->dat[0]              // 1=AEC 2=NLP 3=NS 4=AGC 5=ENC 6=AGC_MAX_LVL
        u8 toggle    = hdl->dat[1]              // 0=关 1=开
        audio_cvp_ioctl(ioctl_cmd, toggle, NULL)

    case CVP_DUT_DEC_DUT_SET (0x13):          // DEC 产测模式
        cvp_dut_hdl->dec_dut_enable = hdl->dat[0]   // 0=关 1=全bypass 2=保留SPK_EQ
        // 下游: audio_dec_dut_en_get(music_decide) 查询此标志

    default: return CVP_DUT_ACK_ERR_UNKNOW (0xA1)

  // 模式切换 → reset_flag=1 → 代码中注释掉了 esco_enc_reset() 调用
  return ret == -1 ? CVP_DUT_ACK_SUCCESS (0x01) : ret
```

## 五、底层 MIC 控制调用链

### 5.1 算法模式 — 双麦/三麦

```c
// audio_aec_dms.c:1135
// 功能：在 AEC 框架内部切换输出通道，不改变 ADC 物理通路

void audio_aec_output_sel(CVP_OUTPUT_ENUM sel, u8 agc)
{
    if (aec_hdl == NULL) return;        // AEC 未初始化 → 命令实际不生效
    if (agc) aec_hdl->attr.EnableBit |= AGC_EN;
    else     aec_hdl->attr.EnableBit &= ~AGC_EN;
    aec_hdl->attr.output_sel = sel;     // DMS_OUTPUT_SEL_MASTER(1) / SLAVE(2) / DEFAULT(0)
}
```

调用来源（均在 `cvp_dut_new_event_deal` 中触发）：

```
CMD 0x01 → audio_aec_output_sel(MASTER, 0)    // 主 MIC 原始数据，关 AGC
CMD 0x02 → audio_aec_output_sel(SLAVE, 0)     // 副 MIC 原始数据，关 AGC
CMD 0x03 → audio_aec_output_sel(DEFAULT, 1)   // 算法全开，AGC 开
CMD 0x0E → audio_aec_output_sel(3, 0)         // FB MIC 原始数据 (三麦)
```

> AGC 在频响测试中必须关闭，否则自动增益会破坏原始频响。

### 5.2 算法模式 — 单麦

```c
// audio_aec.c:704 和 audio_aec_dms.c:825
// 功能：开关整个 CVP 算法链

int audio_aec_toggle_set(u8 toggle)
{
    if (aec_hdl == NULL) return 1;       // AEC 未初始化
    if (toggle) aec_toggle(1);           // 开启 AEC/NLP/ANS/AGC
    else        aec_toggle(0);           // 关闭全部，MIC 原始数据直通
    return 0;
}
```

调用来源：

```
CMD 0x01 → audio_aec_toggle_set(0)   // 单麦关算法
CMD 0x03 → audio_aec_toggle_set(1)   // 单麦开算法
CMD 0x0A → audio_aec_toggle_set(1)   // 全局开 CVP
CMD 0x0B → audio_aec_toggle_set(0)   // 全局关 CVP
```

### 5.3 BYPASS 模式 — ANC MIC

```c
// audio_cvp_dut.c:50
// 功能：直接指定物理 ADC 通道，绕过所有算法

int cvp_dut_bypass_mic_ch_sel(u8 mic_num)
{
    if (cvp_dut_hdl == NULL) return 1;
    switch (mic_num) {
    case A_MIC0: cvp_dut_hdl->mic_ch_sel = AUDIO_ADC_MIC_0; break;
    case A_MIC1: cvp_dut_hdl->mic_ch_sel = AUDIO_ADC_MIC_1; break;
    case A_MIC2: cvp_dut_hdl->mic_ch_sel = AUDIO_ADC_MIC_2; break;
    case A_MIC3: cvp_dut_hdl->mic_ch_sel = AUDIO_ADC_MIC_3; break;
    default: break;
    }
    return 0;
}
```

调用来源：

```
CMD 0x06 (FF MIC)     → cvp_dut_bypass_mic_ch_sel(ANCL_FF_MIC 或 ANCR_FF_MIC)
CMD 0x07 (FB MIC)     → cvp_dut_bypass_mic_ch_sel(ANCL_FB_MIC 或 ANCR_FB_MIC)
CMD 0x08 (右FF MIC)   → cvp_dut_bypass_mic_ch_sel(ANCR_FF_MIC)   仅双耳 ANC
CMD 0x09 (右FB MIC)   → cvp_dut_bypass_mic_ch_sel(ANCR_FB_MIC)   仅双耳 ANC
```

**ANC MIC 与通话 MIC 测试的关键差异**：ANC 命令在通话**之前**发送（BYPASS 模式设置 MIC 通道），进入 HFP 时 `audio_enc_open` 通过 `cvp_dut_mic_ch_get()` 读取选中的通道；通话 MIC 命令在通话**之后**发送（算法模式切换数据源）。

## 六、MIC 校准补偿调用链

> 由 `TCFG_AUDIO_MIC_ARRAY_TRIM_ENABLE` 控制编译。

### 6.1 下发 DIFF 值 (CMD 0x0F)

```
上位机
  计算: DIFF = (主MIC频响 @ 100~1000Hz 均值) - (副MIC频响均值)，单位 dB
  转 float → 十六进制字节
  发送: 0D 0B 01 CC 55 [CRC16] 0F 04 00 [DIFF_float_4B]
          └─外部头──┘ └─magic─┘ └──crc──┘ │ └len─┘ └──参数──┘
                                         CMD=0x0F

设备端 cvp_dut_new_event_deal:
  float diff ← hdl->dat[0..3] (memcpy)
  audio_mic_array_diff_cmp = pow(10, (diff + log10(enc_degradation)*20) / 20)
  syscfg_write(CFG_MIC_ARRAY_DIFF_CMP_VALUE, &cmp, sizeof(float))
    → 写入 Flash VM，持久保存
```

### 6.2 使能补偿 (CMD 0x10)

```
发送: 0A 0B 01 CC 55 [CRC16] 10 01 00 [en]    en=0x01 使能 / 0x00 关闭

设备端:
  app_var.audio_mic_array_trim_en = en
  syscfg_write(CFG_MIC_ARRAY_TRIM_EN, &en, 1) → VM
```

### 6.3 写入金机差值 (CMD 0x14)

```
发送: 15 0B 01 CC 55 [CRC16] 14 0C 00 [talk_diff:4B][ff_diff:4B][fb_diff:4B]

设备端:
  talk_diff ← hdl->dat[0..3], ff_diff ← hdl->dat[4..7], fb_diff ← hdl->dat[8..11]
  app_var.audio_mic_cmp.talk = pow(10, talk_diff/20)
  app_var.audio_mic_cmp.ff   = pow(10, ff_diff/20)
  app_var.audio_mic_cmp.fb   = pow(10, fb_diff/20)
  syscfg_write(CFG_MIC_TARGET_DIFF_CMP, &audio_mic_cmp, sizeof(audio_mic_cmp_t))
```

### 6.4 清除补偿 (CMD 0x11)

```
设备端:
  audio_mic_array_diff_cmp = 1.0f
  audio_mic_cmp.talk = ff = fb = 1.0f
  syscfg_write(CFG_MIC_ARRAY_DIFF_CMP_VALUE, ...)
  syscfg_write(CFG_MIC_TARGET_DIFF_CMP, ...)       // 两个 VM 项都清零
```

**补偿生效机制**：`audio_mic_array_diff_cmp` 和 `audio_mic_cmp.*` 是乘法系数，作用于 MIC 信号幅度。例如主 MIC 灵敏 2dB → `diff_cmp = 10^(2/20) ≈ 1.259` → 副 MIC 数据乘以 1.259 匹配主 MIC。

## 七、完整测试调用序列

以双麦通话 MIC 全通道测试为例，自上而下的完整调用链：

```
① 操作系统启动
     online_spp_init()              注册 SPP 回调
     audio_enc_init() → cvp_dut_init()  注册 DB_PKT_TYPE_DMS 回调

② 上位机连接 SPP
     online_spp_state_cbk(CONNECT) → db_api->init()  激活在线调试

③ 上位机发起 HFP 通话
     [蓝牙协议栈内部回调]
     audio_aec_open(sample_rate, enablebit, out_hdl)  初始化 AEC 框架
       aec_hdl = zalloc(...)
       aec_init(...)

     // ── ANC MIC 测试分支: 时序不同 ──
     // 若测 FF/FB MIC，须在 HFP 建立前发 CMD 0x06~0x09:
     //   cvp_dut_new_event_deal → mode=BYPASS, cvp_dut_bypass_mic_ch_sel()
     //   → 进入 HFP 后 audio_enc_open() 通过 cvp_dut_mic_ch_get() 读取通道
     //   → 直接从指定物理 ADC 通道采集，绕过所有算法

④ 上位机发 CMD 0x01 (关闭算法, 测试 TALK MIC)
     SPP: 09 0B 01 CC 55 30 37 01 00 00
       online_spp_recieve_cbk
         db_api->packet_handle
           db_cb_api_table[0x0B] → cvp_dut_app_online_parse
             cvp_dut_new_rx_packet
               CRC 校验 ✓ → cvp_dut_new_event_deal
                 双麦: audio_aec_output_sel(DMS_OUTPUT_SEL_MASTER, 0)
                 单麦: audio_aec_toggle_set(0)
               cvp_dut_new_tx_packet(0x01)  → SPP 回复
     [上位机播放扫频 → 设备 MIC → ADC → AEC框架(bypass) → SCO上行 → 采集FFT]

⑤ 上位机发 CMD 0x02 (关闭算法, 测试副 MIC)
     SPP: 09 0B 01 CC 55 60 6E 02 00 00
       ...cvp_dut_new_event_deal
         audio_aec_output_sel(DMS_OUTPUT_SEL_SLAVE, 0)
       ...SPP 回复
     [扫频 → 采集 → 计算副 MIC 频响]

⑥ 上位机发 CMD 0x0F (下发 DIFF 值)
     SPP: 0D 0B 01 CC 55 [CRC] 0F 04 00 [float_4B]
       ...cvp_dut_new_event_deal
         memcpy(&diff, hdl->dat, 4)
         pow(10, (diff + dms_mic_diff)/20) → syscfg_write → VM 持久化

⑦ 上位机发 CMD 0x10 (使能补偿)
     SPP: 0A 0B 01 CC 55 B6 3C 10 01 00 01
       ...音频通路中补偿系数生效

⑧ 上位机发 CMD 0x03 (恢复通话算法)
     SPP: 09 0B 01 CC 55 50 59 03 00 00
       ...音频通路恢复正常算法处理

⑨ 上位机发 CMD 0x0D (关机)
     SPP: 09 0B 01 CC 55 51 42 0D 00 00
       sys_enter_soft_poweroff(NULL)
```

## 八、错误处理

### 8.1 回复码

| 回复 CMD | 触发条件 | 代码位置 |
|----------|---------|---------|
| `0x01` (SUCCESS) | 命令识别且执行无异常 | 正常返回 |
| `0xA1` (ERR_UNKNOW) | cmd 不在 switch-case 中 | `cvp_dut_new_event_deal:578` |
| `0xA2` (CRC_ERR) | `head->crc != crc && head->crc != 0x01` | `cvp_dut_new_rx_packet:631` |
| `0xA3` (VM_WRITE_ERR) | `syscfg_write()` 写入长度不匹配 | CMD 0x0F/0x10/0x11/0x14 |

### 8.2 常见静默失败

| 现象 | 原因 | 排查 |
|------|------|------|
| 无任何回复 | `TCFG_AUDIO_CVP_DUT_ENABLE`=0 | 检查 board_cfg.h，确认宏为 1 |
| 无任何回复 | SPP 未连接 (`db_active`=0) | 检查蓝牙 SPP 连接状态 |
| 无任何回复 | MAGIC 不匹配 | payload 前 2B 必须是 CC 55 (LE) |
| 回复成功但无频响数据 | HFP 未建立 (`aec_hdl`=NULL) | 命令执行了但音频通路不存在 |
| ANC MIC 无数据 | 顺序错误：先进入HFP再发ANC命令 | ANC MIC 需先发命令再进通话 |

## 附录 B：CRC16-XModem

```c
// 多项式 0x1021, 初始值 0x0000, 校验范围: CMD+LEN+DATA (不含 MAGIC)
// CRC=0x0001 可跳过校验 (调试用)
u16 CRC16(void *ptr, u32 len) {
    u16 crc = 0x0000;
    u8 *dat = (u8 *)ptr;
    while (len--) {
        crc ^= ((u16)(*dat++)) << 8;
        for (int i = 0; i < 8; i++)
            crc = (crc & 0x8000) ? (crc << 1) ^ 0x1021 : crc << 1;
    }
    return crc;
}
```
