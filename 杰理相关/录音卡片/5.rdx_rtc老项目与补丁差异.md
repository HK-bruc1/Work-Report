# apps\earphone\include\app_config.h

补丁多了关于RTC的时钟源选择

```c
#define RTC_CLK_RES_SEL	            CLK_SEL_32K
```

低功耗掉电ram由2变3

```
#define TCFG_LOWPOWER_RAM_SIZE				3	                // 低功耗掉电ram大小，单位：128K，可设置值：0、2、3
```

# apps\earphone\device_config.c

有关RTC的接口换为新的头文件

```c
#include "rtc/rtc_dev.h"
代替
#include "asm/rtc.h"
```

替换一下宏相关的定义

```c
/* extern void alarm_isr_user_cbfun(u8 index); */
RTC_DEV_PLATFORM_DATA_BEGIN(rtc_data)
.default_sys_time = &def_sys_time,
 .default_alarm = &def_alarm,

  .rtc_clk = RTC_CLK_RES_SEL,//引用板级配置中的宏定义，默认也是CLK_SEL_32K
   .rtc_sel = HW_RTC,
    //闹钟中断的回调函数,用户自行定义
    .cbfun = NULL,
     /* .cbfun = alarm_isr_user_cbfun, */
     RTC_DEV_PLATFORM_DATA_END()

代替

/* extern void alarm_isr_user_cbfun(u8 index); */
RTC_DEV_PLATFORM_DATA_BEGIN(rtc_data)
.default_sys_time = &def_sys_time,
 .default_alarm = &def_alarm,

  .clk_sel = CLK_SEL_32K,
   /* #if defined(CONFIG_CPU_BR27) */
   /* .rtc_sel = HW_RTC, */
   /* #endif */
   //闹钟中断的回调函数,用户自行定义
   .cbfun = NULL,
    /* .cbfun = alarm_isr_user_cbfun, */
    RTC_DEV_PLATFORM_DATA_END()         
```

# rtc_dev.h与rtc.h

新补丁中多了很多接口，但是接口又是封装的，那么换库cpu.a就顺利成章了，静态库中没有的话，链接阶段就会报错。

# interface\utils\syscfg_id.h

VM中的地址编号搞两个给RTC使用

```c
#define     VM_RTC_ALM_TIME             105
#define     VM_RTC_TIME_TRIM            106

代替

#define     VM_OLD_RTC_TIME             105
#define     VM_OLD_REAL_TIME            106
```
